100  // External Interrupts
101  // Warning: This will disable any previously enabled interrupt which is
102  // routed to the same interupt line, e.g. PA3 will disable PB3.
103  // This is a hardware limitation by the STM32 EXTI.
104  inline static void enableExternalInterrupt()
105  {
106  // PA[x], x = 0 .. 3 maps to EXTICR[0]
107  // PA[x], x = 4 .. 7 maps to EXTICR[1]
108  // PA[x], x = 8 .. 11 maps to EXTICR[2]
109  // PA[x], x = 12 .. 15 maps to EXTICR[3]
110  // => bit3 and bit2 (mask 0x0c) specify the register
111  // => bit1 and bit0 (mask 0x03) specify the bit position
112  constexpr uint8_t index = (pin & 0b1100) >> 2;
113  constexpr uint8_t bit_pos = (pin & 0b0011) << 2;
114  constexpr uint16_t syscfg_mask = (0b1111) << bit_pos;
115  constexpr uint16_t syscfg_value = (port_nr & (0b1111)) << bit_pos;
116  SYSCFG->EXTICR[index] = (SYSCFG->EXTICR[index] & ~syscfg_mask) | syscfg_value;
117  EXTI->IMR |= mask;
118  }
119  inline static void disableExternalInterrupt() { EXTI->IMR &= ~mask; }
120  inline static void enableExternalInterruptVector(const uint32_t priority)
121  {

124  }
