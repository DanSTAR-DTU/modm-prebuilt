221 #define RF_ISEMPTY(...) _RF_ISEMPTY( \
222  /* test if there is just one argument, eventually an empty one */ \
223  RF_HAS_COMMA(__VA_ARGS__), \
224  /* test if _TRIGGER_PARENTHESIS_ together with the argument adds a comma */ \
225  RF_HAS_COMMA(RF_TRIGGER_PARENTHESIS_ __VA_ARGS__), \
226  /* test if the argument together with a parenthesis adds a comma */ \
227  RF_HAS_COMMA(__VA_ARGS__ (/*empty*/)), \
228  /* test if placing it between _TRIGGER_PARENTHESIS_ and the parenthesis adds a comma */ \
