149  template<class SystemClock>
150  static Value
151  setPeriod(uint32_t microseconds, bool autoApply = true)
152  {
153  // This will be inaccurate for non-smooth frequencies (last six digits
154  // unequal to zero)
155  uint32_t cycles = microseconds * (SystemClock::Timer11 / 1000000UL);
156  uint16_t prescaler = (cycles + 65535) / 65536; // always round up
157  Value overflow = cycles / prescaler;
158 
159  overflow = overflow - 1; // e.g. 36000 cycles are from 0 to 35999
160 
161  setPrescaler(prescaler);
162  setOverflow(overflow);
163 
164  if (autoApply) {
165  // Generate Update Event to apply the new settings for ARR
166  TIM11->EGR |= TIM_EGR_UG;
167  }
168 
169  return overflow;
170  }

209  {

213  uint32_t flags = static_cast<uint32_t>(OutputCompareMode::Pwm2);
214 
215  if (channel <= 1)
