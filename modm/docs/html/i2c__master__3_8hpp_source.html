70  template<class SystemClock, baudrate_t baudrate=kBd(100), percent_t tolerance=pct(5)>
71  static modm_always_inline void
72  initialize()
73  {
74  // calculate the expected clock ratio
75  constexpr uint8_t scalar = (baudrate <= 100000) ? 2 : ((baudrate <= 300000) ? 3 : 25);
76 
77  // calculate the fractional prescaler value
78  constexpr float pre_raw = float(SystemClock::I2c3) / (scalar * baudrate);
79  // respect the prescaler range of 1 or 4 to 4095
80  constexpr uint32_t pre_ceil = std::ceil(pre_raw) > 4095 ? 4095 : std::ceil(pre_raw);
81  constexpr uint32_t pre_floor = std::floor(pre_raw) < ((scalar < 3) ? 4 : 1) ?
82  ((scalar < 3) ? 4 : 1) : std::floor(pre_raw);
83 
84  // calculate the possible baudrates above and below the requested baudrate
85  constexpr uint32_t baud_lower = SystemClock::I2c3 / ( scalar * pre_ceil );
86  constexpr uint32_t baud_upper = SystemClock::I2c3 / ( scalar * pre_floor );
87 
88  // calculate the fractional prescaler value corresponding to the baudrate exactly
89  // between the upper and lower baudrate
90  constexpr uint32_t baud_middle = (baud_upper + baud_lower) / 2;
91  // decide which prescaler value is closer to a possible baudrate
92  constexpr uint32_t pre = (baudrate < baud_middle) ? pre_ceil : pre_floor;
93 
94  // check if within baudrate tolerance
95  constexpr uint32_t generated_baudrate = SystemClock::I2c3 / ( scalar * pre );
96  assertBaudrateInTolerance<
97  /* clostest available baudrate */ generated_baudrate,
98  /* desired baudrate */ baudrate,
99  tolerance >();
100 
101  // the final prescaler value is augmented with the F/S and DUTY bit.
102  constexpr uint32_t prescaler = pre |
103  ((scalar >= 3) ? (1 << 15) : 0) |
104  ((scalar == 25) ? (1 << 14) : 0);
105 
106  // peripheral frequency clock
107  constexpr uint8_t freq = SystemClock::I2c3 / 1000000;
108 
109  // maximum rise time: assuming its linear:
110  // 1000ns @ 100kHz and 300ns @ 400kHz
111  // => y = x * m + b, with m = -2.3333ns/kHz, b = 1233.3333ns
112  constexpr float max_rise_time = -2.333333f * (float(baudrate) / 1000.f) + 1233.333333f;
113  // calculate trise
114  constexpr float trise_raw = max_rise_time < 0 ? 0 : std::floor(max_rise_time / (1000.f / freq));
115  constexpr uint8_t trise = trise_raw > 62 ? 63 : (trise_raw + 1);
116 
117  initializeWithPrescaler(freq, trise, prescaler);
118  }
