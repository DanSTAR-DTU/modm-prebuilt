36 void
37 DataDouble::calculateCalibratedTemperature()
38 {
39  // Temperature value is expected to be received in 20 bit format,
40  // positive, stored in a 32 bit signed integer.
41  int32_t adc = ((raw[3] << 16) | (raw[4] << 8) | (raw[5] << 0));
42  adc >>= 4;
43 
44  double adc_16384 = double(adc) / double(16384.0);
45  double adc_131072 = double(adc) / double(131072.0);
46 
47  double T1_1024 = double(calibration.T1) / double(1024.0);
48  double T1_8192 = double(calibration.T1) / double(8192.0);
49  double T2 = double(calibration.T2);
50  double T3 = double(calibration.T3);
51 
52  // double var1, var2, T;
53  double var1 = (adc_16384 - T1_1024) * T2;
54  double var2 = ((adc_131072 - T1_8192) * (adc_131072 - T1_8192)) * T3;
55 
56  t_fine = var1 + var2;
57 
58  calibratedTemperatureDouble = (var1 + var2) / double(5120.0);
59 
60  MODM_LOG_DEBUG.printf("T dp = %4.2f\n", calibratedTemperatureDouble);
61 
62  meta |= TEMPERATURE_CALCULATED;
63 }

65 void
66 DataDouble::calculateCalibratedPressure()
67 {
68  // Before pressure can be calculated, temperature must be.
69  if (not (meta & TEMPERATURE_CALCULATED)) {
70  calculateCalibratedTemperature();
71  }
72 
73  int32_t adc = (((int32_t(raw[0])) << 16) | (raw[1] << 8) | (raw[2] << 0));
74  adc >>= 4;
75 
76  double P1 = double(calibration.P1);
77  double P2 = double(calibration.P2);
78  double P3 = double(calibration.P3);
79  double P4 = double(calibration.P4);
80  double P5 = double(calibration.P5);
81  double P6 = double(calibration.P6);
82  double P7 = double(calibration.P7);
83  double P8 = double(calibration.P8);
84  double P9 = double(calibration.P9);
85 
86  double var1 = (t_fine / double(2.0)) - double(64'000.0);
87 
88  double var2 = var1 * var1 * P6 / double(32'768.0);
89  var2 = var2 + var1 * P5 * double(2.0);
90  var2 = var2 / double(4.0) + (P4 * double(65'636.0));
91 
92  var1 = (P3 * var1 * var1 / double(524'288.0) + P2 * var1) / double(524'288.0);
93  var1 = (double(1.0) + var1 / double(32'768.0)) * P1;
94  if (var1 == double(0.0)) {
95  calibratedPressureDouble = double(0.0);
96  return;
97  }
98 
99  double p = double(1'048'576.0) - double(adc);
100  p = (p - (var2 / double(4'096.0))) * double(6'250.0) / var1;
101 
102  var1 = P9 * p * p / double(2'147'483'648.0);
103  var2 = p * P8 / double(32'768.0);
104 
105  p = p + (var1 + var2 + P7) / double(16.0);
106 
107  calibratedPressureDouble = p;
108 
109  meta |= PRESSURE_CALCULATED;
110 }
