58 template<int32_t Clk, int32_t Bitrate, uint8_t prescaler_width = 10, uint8_t bs1_width = 4, uint8_t bs2_width = 3, uint8_t sjw_width = 2>
59 class CanBitTiming
60 {
61 private:
62  static constexpr uint32_t round_uint32(float f)
63  {
64  uint32_t f_int = (uint32_t) f;
65  if(f - f_int > 0.5)
66  return f_int + 1;
67  else
68  return f_int;
69  }
70 
71  static constexpr CanBitTimingConfiguration calculateBestConfig()
72  {
73  constexpr uint8_t minBs1Bs2 = 14;
74  constexpr uint8_t maxBs1Bs2 = 20;
75 
76  float minError = 10'000.0;
77  uint16_t bestPrescaler = 0;
78  uint8_t bestBs1Bs2 = 0;
79 
80  for(uint8_t bs1Bs2 = minBs1Bs2; bs1Bs2 <= maxBs1Bs2; ++bs1Bs2) {
81  float idealPrescaler = float(Clk) / (Bitrate * (1 + bs1Bs2));
82  uint32_t intPrescaler = round_uint32(idealPrescaler);
83 
84  if(intPrescaler < ((1 << prescaler_width) - 1)) {
85  float error = constexpr_fabs(1 - intPrescaler/idealPrescaler);
86  if(error <= minError) {
87  bestPrescaler = intPrescaler;
88  minError = error;
89  bestBs1Bs2 = bs1Bs2;
90  }
91  }
92  }
93 
94  uint8_t bs2 = round_uint32(0.275f * (bestBs1Bs2 + 1));
95  uint8_t bs1 = bestBs1Bs2 - bs2;
96 
97  return CanBitTimingConfiguration{bs1, bs2, 1, bestPrescaler, minError};
98  }
