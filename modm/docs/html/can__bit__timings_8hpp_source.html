68  static constexpr CanBitTimingConfiguration calculateBestConfig()

73  float minError = 10'000.0;

77  for(uint8_t bs1Bs2 = minBs1Bs2; bs1Bs2 <= maxBs1Bs2; ++bs1Bs2) {
78  float idealPrescaler = float(Clk) / (Bitrate * (1 + bs1Bs2));
79  uint32_t intPrescaler = round_uint32(idealPrescaler);
