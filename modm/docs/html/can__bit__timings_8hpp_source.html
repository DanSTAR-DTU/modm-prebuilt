59  static constexpr uint32_t round_uint32(float f)
60  {
61  uint32_t f_int = (uint32_t) f;
62  if(f - f_int > 0.5)
63  return f_int + 1;
64  else
65  return f_int;
66  }

68  static constexpr CanBitTimingConfiguration calculateBestConfig()
69  {
70  constexpr uint8_t minBs1Bs2 = 14;
71  constexpr uint8_t maxBs1Bs2 = 20;
72 
73  float minError = 10000;
74  uint16_t bestPrescaler = 0;
75  uint8_t bestBs1Bs2 = 0;
76 
77  for(uint8_t bs1Bs2 = minBs1Bs2; bs1Bs2 <= maxBs1Bs2; ++bs1Bs2) {
78  float idealPrescaler = float(Clk) / (Bitrate * (1 + bs1Bs2));
79  uint32_t intPrescaler = round_uint32(idealPrescaler);
80  float error = fabs(1 - intPrescaler/idealPrescaler);
81  if(error <= minError) {
82  bestPrescaler = intPrescaler;
83  minError = error;
84  bestBs1Bs2 = bs1Bs2;
85  }
86  }
87 
88  uint8_t bs2 = round_uint32(0.275f * (bestBs1Bs2 + 1));
89  uint8_t bs1 = bestBs1Bs2 - bs2;
90 
91  return CanBitTimingConfiguration{bs1, bs2, 1, bestPrescaler, minError};
92  }
