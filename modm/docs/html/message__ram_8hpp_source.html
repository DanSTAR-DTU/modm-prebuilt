182  static void
183  readData(RxFifoAddress address, std::span<uint8_t> outData)
184  {
185  const auto size = std::min(outData.size(), 64u);
186 
187  // + 2: skip 2x32 bit headers
188  const uint32_t* messageRam = address.ptr() + 2;
189  // message data must be read in 32bit accesses, memcpy on message ram does not work
190  for (auto i = 0u; i < size; i += 4) {
191  const uint32_t receivedData = *messageRam++;
192 
193  // copy in 32 bit words, memcpy is optimized to single store instruction
194  // CAN message buffer must fit full multiples of 4 bytes
195  std::memcpy(&outData[i], &receivedData, 4);
196  }
197  }
