154  template<class SystemClock>
155  static Value
156  setPeriod(uint32_t microseconds, bool autoApply = true)
157  {
158  // This will be inaccurate for non-smooth frequencies (last six digits
159  // unequal to zero)
160  uint32_t cycles = microseconds * (SystemClock::Timer5 / 1'000'000UL);
161  uint16_t prescaler = (cycles + 65'535) / 65'536; // always round up
162  Value overflow = cycles / prescaler;
163 
164  overflow = overflow - 1; // e.g. 36'000 cycles are from 0 to 35'999
165 
166  setPrescaler(prescaler);
167  setOverflow(overflow);
168 
169  if (autoApply) {
170  // Generate Update Event to apply the new settings for ARR
171  TIM5->EGR |= TIM_EGR_UG;
172  }
173 
174  return overflow;
175  }

257  /*
258  * Set Dead Time Value
259  *
260  * Different Resolution Depending on DeadTime[7:5]:
261  * 0xx => DeadTime[6:0] * T(DTS)
262  * 10x => (DeadTime[5:0] + 32) * 2 * T(DTS)
263  * 110 => (DeadTime[4:0] + 4) * 8 * T(DTS)
264  * 111 => (DeadTime[4:0] + 2) * 16 * T(DTS)
265  */
266  static inline void
267  setDeadTime(DeadTimeResolution resolution, uint8_t deadTime)
268  {
269  uint8_t bitmask;
270  switch(resolution){
271  case DeadTimeResolution::From0With125nsStep:
272  bitmask = 0b01111111;
273  break;
274  case DeadTimeResolution::From16usWith250nsStep:
275  bitmask = 0b00111111;
276  break;
277  case DeadTimeResolution::From32usWith1usStep:
278  case DeadTimeResolution::From64usWith2usStep:
279  bitmask = 0b00011111;
280  break;
281  default:
282  bitmask = 0x00;
283  break;
284  }
285  uint32_t flags = TIM5->BDTR;
286  flags &= ~TIM_BDTR_DTG;
287  flags |= (deadTime & bitmask) | static_cast<uint32_t>(resolution);
288  TIM5->BDTR = flags;
289  }
