153  template<class SystemClock>
154  static Value
155  setPeriod(uint32_t microseconds, bool autoApply = true)
156  {
157  // This will be inaccurate for non-smooth frequencies (last six digits
158  // unequal to zero)
159  uint32_t cycles = microseconds * (SystemClock::Timer5 / 1'000'000UL);
160  uint16_t prescaler = (cycles + 65'535) / 65'536; // always round up
161  Value overflow = cycles / prescaler;
162 
163  overflow = overflow - 1; // e.g. 36'000 cycles are from 0 to 35'999
164 
165  setPrescaler(prescaler);
166  setOverflow(overflow);
167 
168  if (autoApply) {
169  // Generate Update Event to apply the new settings for ARR
170  TIM5->EGR |= TIM_EGR_UG;
171  }
172 
173  return overflow;
174  }

256  /*
257  * Set Dead Time Value
258  *
259  * Different Resolution Depending on DeadTime[7:5]:
260  * 0xx => DeadTime[6:0] * T(DTS)
261  * 10x => (DeadTime[5:0] + 32) * 2 * T(DTS)
262  * 110 => (DeadTime[4:0] + 4) * 8 * T(DTS)
263  * 111 => (DeadTime[4:0] + 2) * 16 * T(DTS)
264  */
265  static inline void
266  setDeadTime(DeadTimeResolution resolution, uint8_t deadTime)
267  {
268  uint8_t bitmask;
269  switch(resolution){
270  case DeadTimeResolution::From0With125nsStep:
271  bitmask = 0b01111111;
272  break;
273  case DeadTimeResolution::From16usWith250nsStep:
274  bitmask = 0b00111111;
275  break;
276  case DeadTimeResolution::From32usWith1usStep:
277  case DeadTimeResolution::From64usWith2usStep:
278  bitmask = 0b00011111;
279  break;
280  default:
281  bitmask = 0x00;
282  break;
283  }
284  uint32_t flags = TIM5->BDTR;
285  flags &= ~TIM_BDTR_DTG;
286  flags |= (deadTime & bitmask) | static_cast<uint32_t>(resolution);
287  TIM5->BDTR = flags;
288  }
