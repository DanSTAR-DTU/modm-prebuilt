149  template<class SystemClock>
150  static Value
151  setPeriod(uint32_t microseconds, bool autoApply = true)
152  {
153  // This will be inaccurate for non-smooth frequencies (last six digits
154  // unequal to zero)
155  uint32_t cycles = microseconds * (SystemClock::Timer13 / 1'000'000UL);
156  uint16_t prescaler = (cycles + 65'535) / 65'536; // always round up
157  Value overflow = cycles / prescaler;
158 
159  overflow = overflow - 1; // e.g. 36'000 cycles are from 0 to 35'999
160 
161  setPrescaler(prescaler);
162  setOverflow(overflow);
163 
164  if (autoApply) {
165  // Generate Update Event to apply the new settings for ARR
166  TIM13->EGR |= TIM_EGR_UG;
167  }
168 
169  return overflow;
170  }

207  static void
208  setInvertedPwm(uint32_t channel)
209  {
210  channel -= 1; // 1..2 -> 0..1
211 
212  {
213  uint32_t flags = static_cast<uint32_t>(OutputCompareMode::Pwm2);
214 
215  if (channel <= 1)
216  {
217  uint32_t offset = 8 * channel;
218 
219  flags <<= offset;
220  flags |= TIM13->CCMR1 & ~(TIM_CCMR1_OC1M << offset);
221  TIM13->CCMR1 = flags;
222  }
223  else {
224  uint32_t offset = 8 * (channel - 2);
225 
226  flags <<= offset;
227  flags |= TIM13->CCMR2 & ~(TIM_CCMR1_OC1M << offset);
228 
229  TIM13->CCMR2 = flags;
230  }
