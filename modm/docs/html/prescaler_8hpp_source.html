136  const uint32_t prescaler_ceiling = std::min(uint32_t(std::ceil(desired)), end);

140  if (desired_frequency <= baud_middle)
141  return {input_frequency, desired_frequency, prescaler_ceiling - begin, prescaler_ceiling};
142  return {input_frequency, desired_frequency, prescaler_floor - begin, prescaler_floor};
