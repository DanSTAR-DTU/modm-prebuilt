143  template<class SystemClock>
144  static uint16_t
145  setPeriod(uint32_t microseconds, bool autoApply = true)
146  {
147  // This will be inaccurate for non-smooth frequencies (last six digits
148  // unequal to zero)
149  uint32_t cycles = microseconds * (SystemClock::Timer1 / 1'000'000UL);
150  uint16_t prescaler = (cycles + 65'535) / 65'536; // always round up
151  uint16_t overflow = cycles / prescaler;
152 
153  overflow = overflow - 1; // e.g. 36'000 cycles are from 0 to 35'999
154 
155  setPrescaler(prescaler);
156  setOverflow(overflow);
157 
158  if (autoApply) {
159  // Generate Update Event to apply the new settings for ARR
160  TIM1->EGR |= TIM_EGR_UG;
161  }
162 
163  return overflow;
164 }

238  /*
239  * Set Dead Time Value
240  *
241  * Different Resolution Depending on DeadTime[7:5]:
242  * 0xx => DeadTime[6:0] * T(DTS)

256  /*
257  * Set Dead Time Value
258  *
259  * Different Resolution Depending on DeadTime[7:5]:
260  * 0xx => DeadTime[6:0] * T(DTS)
261  * 10x => (DeadTime[5:0] + 32) * 2 * T(DTS)
262  * 110 => (DeadTime[4:0] + 4) * 8 * T(DTS)
263  * 111 => (DeadTime[4:0] + 2) * 16 * T(DTS)
264  */
265  static inline void
266  setDeadTime(DeadTimeResolution resolution, uint8_t deadTime)
267  {
268  uint8_t bitmask;
269  switch(resolution){
270  case DeadTimeResolution::From0With125nsStep:
271  bitmask = 0b01111111;
272  break;
273  case DeadTimeResolution::From16usWith250nsStep:
274  bitmask = 0b00111111;
275  break;
276  case DeadTimeResolution::From32usWith1usStep:
277  case DeadTimeResolution::From64usWith2usStep:
278  bitmask = 0b00011111;
279  break;
280  default:
281  bitmask = 0x00;
282  break;
283  }
284  uint32_t flags = TIM1->BDTR;
285  flags &= ~TIM_BDTR_DTG;
286  flags |= (deadTime & bitmask) | static_cast<uint32_t>(resolution);
287  TIM1->BDTR = flags;
288  }
