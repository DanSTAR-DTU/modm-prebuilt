131  template<class SystemClock>
132  static uint16_t
133  setPeriod(uint32_t microseconds, bool autoApply = true)
134  {
135  // This will be inaccurate for non-smooth frequencies (last six digits
136  // unequal to zero)
137  uint32_t cycles = microseconds * (SystemClock::Timer20 / 1'000'000UL);
138  uint16_t prescaler = (cycles + 65'535) / 65'536; // always round up
139  uint16_t overflow = cycles / prescaler;
140 
141  overflow = overflow - 1; // e.g. 36'000 cycles are from 0 to 35'999
142 
143  setPrescaler(prescaler);
144  setOverflow(overflow);
145 
146  if (autoApply) {
147  // Generate Update Event to apply the new settings for ARR
148  TIM20->EGR |= TIM_EGR_UG;
149  }
150 
151  return overflow;
152 }

226  /*
227  * Set Dead Time Value
228  *
229  * Different Resolution Depending on DeadTime[7:5]:
230  * 0xx => DeadTime[6:0] * T(DTS)
231  * 10x => (DeadTime[5:0] + 32) * 2 * T(DTS)
232  * 110 => (DeadTime[4:0] + 4) * 8 * T(DTS)
233  * 111 => (DeadTime[4:0] + 2) * 16 * T(DTS)

244  /*
245  * Set Dead Time Value
246  *
247  * Different Resolution Depending on DeadTime[7:5]:
248  * 0xx => DeadTime[6:0] * T(DTS)
249  * 10x => (DeadTime[5:0] + 32) * 2 * T(DTS)
250  * 110 => (DeadTime[4:0] + 4) * 8 * T(DTS)
251  * 111 => (DeadTime[4:0] + 2) * 16 * T(DTS)
252  */
253  static inline void
254  setDeadTime(DeadTimeResolution resolution, uint8_t deadTime)
255  {
256  uint8_t bitmask;
257  switch(resolution){
258  case DeadTimeResolution::From0With125nsStep:
259  bitmask = 0b01111111;
260  break;
261  case DeadTimeResolution::From16usWith250nsStep:
262  bitmask = 0b00111111;
263  break;
264  case DeadTimeResolution::From32usWith1usStep:
265  case DeadTimeResolution::From64usWith2usStep:
266  bitmask = 0b00011111;
267  break;
268  default:
269  bitmask = 0x00;
270  break;
271  }
272  uint32_t flags = TIM20->BDTR;
273  flags &= ~TIM_BDTR_DTG;
274  flags |= (deadTime & bitmask) | static_cast<uint32_t>(resolution);
275  TIM20->BDTR = flags;
276  }
