151  template<class SystemClock>
152  static Value
153  setPeriod(uint32_t microseconds, bool autoApply = true)
154  {
155  // This will be inaccurate for non-smooth frequencies (last six digits
156  // unequal to zero)
157  uint32_t cycles = microseconds * (SystemClock::Timer4 / 1'000'000UL);
158  uint16_t prescaler = (cycles + 65'535) / 65'536; // always round up
159  Value overflow = cycles / prescaler;
160 
161  overflow = overflow - 1; // e.g. 36'000 cycles are from 0 to 35'999
162 
163  setPrescaler(prescaler);
164  setOverflow(overflow);
165 
166  if (autoApply) {
167  // Generate Update Event to apply the new settings for ARR
168  TIM4->EGR |= TIM_EGR_UG;
169  }
170 
171  return overflow;
172  }

209  static void
210  setInvertedPwm(uint32_t channel)
211  {
212  channel -= 1; // 1..2 -> 0..1
213 
214  {
215  uint32_t flags = static_cast<uint32_t>(OutputCompareMode::Pwm2);
216 
217  if (channel <= 1)
218  {
219  uint32_t offset = 8 * channel;
220 
221  flags <<= offset;
222  flags |= TIM4->CCMR1 & ~(TIM_CCMR1_OC1M << offset);
223  TIM4->CCMR1 = flags;
224  }
225  else {
226  uint32_t offset = 8 * (channel - 2);
227 
228  flags <<= offset;
229  flags |= TIM4->CCMR2 & ~(TIM_CCMR1_OC1M << offset);
230 
231  TIM4->CCMR2 = flags;
232  }
