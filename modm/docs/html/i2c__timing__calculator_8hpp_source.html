55  explicit constexpr I2cTimingCalculator(const I2cParameters& parameters)

58  constexpr std::optional<I2cMasterTimings>
59  calculateTimings() const
60  {
61  uint16_t prescalerMask{findValidPrescalers()};
62  if(prescalerMask == 0) {
63  return std::nullopt;
64  }
65 
66  if((params.fallTime > ModeConstants::tFallMax[modeIndex()])
67  || (params.riseTime > ModeConstants::tRiseMax[modeIndex()])) {
68  return std::nullopt;
69  }
70 
71  bool found = false;
72  float lastError = 1e10;
73  uint8_t bestSclLow = 0;
74  uint8_t bestSclHigh = 0;
75  uint8_t bestPrescaler = 16;
76 
77  for(int prescaler = 15; prescaler >= 0; --prescaler)
78  {
79  if(not (prescalerMask & (1 << prescaler))) {
80  continue;
81  }
82 
83  auto [valid, sclLowMin, sclHighMin] = minimumSclLowHigh(prescaler);
84  if(!valid) {
85  continue;
86  }
87 
88  const uint16_t sclSumMax = maximumSclSum(prescaler);
89  const uint16_t sclLowMax = std::min<uint16_t>(sclSumMax, 255);
90 
91  for(uint16_t sclLow = sclLowMin; sclLow <= sclLowMax; ++sclLow) {
92  auto sclHighMax = std::min(sclSumMax - sclLow, 255);
93 
94  auto sclHigh = findBestSclHigh(prescaler, sclLow, sclHighMin, sclHighMax);
95  float speed = calculateSpeed(prescaler, sclLow, sclHigh);
96  auto error = std::fabs((speed / params.targetSpeed) - 1.0f);
97  // modm::Tolerance value is in unit [1/1000]
98  if ((error < pct2f(params.tolerance)) && (error <= lastError) && (prescaler <= bestPrescaler)) {
99  lastError = error;
100  bestPrescaler = prescaler;
101  bestSclLow = sclLow;
102  bestSclHigh = sclHigh;
103  found = true;
104  }
105  }
106  }
107 
108  if(found) {
109  I2cMasterTimings timings = {};
110  timings.prescaler = bestPrescaler;
111 
112  auto [sdaDel, sclDel] = findHoldTimeSettings(bestPrescaler).value();
113  timings.sdaDel = sdaDel;
114  timings.sclDel = sclDel;
115 
116  timings.sclLow = bestSclLow;
117  timings.sclHigh = bestSclHigh;
118 
119  return timings;
120  }
121 
122  return std::nullopt;
123  }

125  static constexpr uint32_t
126  timingsToRegisterValue(const I2cMasterTimings& timings)
127  {
128  uint32_t value = timings.sclLow;
129  value |= timings.sclHigh << 8;
130  value |= (timings.sdaDel & 0b1111) << 16;
131  value |= (timings.sclDel & 0b1111) << 20;
132  value |= (timings.prescaler & 0b1111) << 28;
133  return value;
134  }

206  constexpr std::optional<std::tuple<uint16_t, uint16_t>>
207  findHoldTimeSettings(uint8_t prescaler) const
208  {
209  auto sdaDelMinTime = params.fallTime - AnalogFilterDelayMin
210  - (float(params.digitalFilterLength + 3) / params.peripheralClock);
211 
212  sdaDelMinTime = std::max(0.0f, sdaDelMinTime);
213 
214  auto sdaDelMaxTime = ModeConstants::tHdDatMax[modeIndex()];
215  sdaDelMaxTime -= params.riseTime;
216  sdaDelMaxTime -= (float(params.digitalFilterLength + 4) / params.peripheralClock);
217  sdaDelMaxTime -= AnalogFilterDelayMax;
218 
219  sdaDelMaxTime = std::max(0.0f, sdaDelMaxTime);
220 
221  auto sclDelMinTime = params.riseTime + ModeConstants::tSuDatMin[modeIndex()];
222 
223  for(uint8_t sdaDel = 0; sdaDel <= 15; ++sdaDel) {
224  auto sdaDelTime = float(sdaDel) * (prescaler + 1) / params.peripheralClock;
225 
226  if(sdaDelTime < sdaDelMinTime || sdaDelTime > sdaDelMaxTime) {
227  continue;
228  }
229 
230  for(uint8_t sclDel = 0; sclDel <= 15; ++sclDel) {
231  auto sclDelTime = float(sclDel + 1) * (prescaler + 1) / params.peripheralClock;
232 
233  if(sclDelTime >= sclDelMinTime) {
234  return std::tuple{sdaDel, sclDel};
235  }
236  }
237  }
238 
239  return std::nullopt;
240  }

258  constexpr std::tuple<bool, uint8_t, uint8_t>
259  minimumSclLowHigh(uint8_t prescaler) const
260  {
261  auto clockPeriod = float(prescaler + 1) / params.peripheralClock;
262 
263  float lowMinFloat = std::max(
264  ((ModeConstants::tLowMin[modeIndex()] - SyncTime) / clockPeriod) - 1,
265  ((4.0f / params.peripheralClock) + FilterDelay - SyncTime) / clockPeriod - 1
266  );
267 
268  float highMinFloat = std::max(
269  ((ModeConstants::tHighMin[modeIndex()] - SyncTime) / clockPeriod) - 1,
270  ((1.0f / params.peripheralClock) - SyncTime) / clockPeriod - 1
271  );
272 
273  lowMinFloat = std::ceil(lowMinFloat);
274  highMinFloat = std::ceil(highMinFloat);
275 
276  if(lowMinFloat > 255 || highMinFloat > 255) {
277  return {false, 255, 255};
278  }
279 
280  uint8_t lowMin = (uint8_t) std::max(lowMinFloat, 0.f);
281  uint8_t highMin = (uint8_t) std::max(highMinFloat, 0.f);
282  return {true, lowMin, highMin};
283  }

285  constexpr uint16_t
286  maximumSclSum(uint8_t prescaler) const
287  {
288  auto clockPeriod = float(prescaler + 1) / params.peripheralClock;
289 
290  auto targetSclTime = 1.f / params.targetSpeed;
291  auto sclTimeMax = targetSclTime * (1.f + pct2f(params.tolerance));
292  auto maxSclSum = ((sclTimeMax - params.riseTime - params.fallTime - 2*SyncTime)
293  / clockPeriod) - 2;
294 
295  return (uint16_t) std::max(0.0f, std::min(maxSclSum, 255.0f * 2));
296  }

298  constexpr uint8_t
299  findBestSclHigh(uint8_t prescaler, uint8_t sclLow, uint8_t min, uint8_t max) const
300  {
301  auto clockPeriod = float(prescaler + 1) / params.peripheralClock;
302  auto sclLowTime = (sclLow + 1) * clockPeriod + SyncTime;
303 
304  auto targetSclTime = 1.f / params.targetSpeed;
305  auto targetSclHighTime = targetSclTime - sclLowTime
306  - params.riseTime - params.fallTime;
307 
308  auto targetSclHigh = std::round((targetSclHighTime - SyncTime) / clockPeriod - 1);
309 
310  return (uint8_t) std::max<float>(min, std::min<float>(targetSclHigh, max));
311  }
312 
313  constexpr float
314  calculateSpeed(uint8_t prescaler, uint16_t sclLow, uint16_t sclHigh) const
315  {
316  auto clockPeriod = float(prescaler + 1) / params.peripheralClock;
317 
318  auto sclLowTime = (sclLow + 1) * clockPeriod + SyncTime;
319  auto sclHighTime = (sclHigh + 1) * clockPeriod + SyncTime;
320 
321  auto sclTime = sclLowTime + sclHighTime + params.riseTime + params.fallTime;
322  auto speed = 1.0f / sclTime;
323 
324  return speed;
325  }
