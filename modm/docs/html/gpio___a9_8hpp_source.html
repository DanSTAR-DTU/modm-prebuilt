97  // External Interrupts
98  // Warning: This will disable any previously enabled interrupt which is
99  // routed to the same interupt line, e.g. PA3 will disable PB3.
100  // This is a hardware limitation by the STM32 EXTI.
101  inline static void enableExternalInterrupt()
102  {
103  // PA[x], x = 0 .. 3 maps to EXTICR[0]
104  // PA[x], x = 4 .. 7 maps to EXTICR[1]
105  // PA[x], x = 8 .. 11 maps to EXTICR[2]
106  // PA[x], x = 12 .. 15 maps to EXTICR[3]
107  // => bit3 and bit2 (mask 0x0c) specify the register
108  // => bit1 and bit0 (mask 0x03) specify the bit position
109  constexpr uint8_t index = (pin & 0b1100) >> 2;
110  constexpr uint8_t bit_pos = (pin & 0b0011) << 2;
111  constexpr uint16_t syscfg_mask = (0b1111) << bit_pos;
112  constexpr uint16_t syscfg_value = (port_nr & (0b1111)) << bit_pos;
113  SYSCFG->EXTICR[index] = (SYSCFG->EXTICR[index] & ~syscfg_mask) | syscfg_value;
114  EXTI->IMR |= mask;
115  }
116  inline static void disableExternalInterrupt() { EXTI->IMR &= ~mask; }
117  inline static void enableExternalInterruptVector(const uint32_t priority)
118  {
119  NVIC_SetPriority(ExternalInterruptIRQ, priority);
120  NVIC_EnableIRQ(ExternalInterruptIRQ);
121  }
122  inline static void disableExternalInterruptVector() { NVIC_DisableIRQ(ExternalInterruptIRQ); }
123  inline static void setInputTrigger(const InputTrigger trigger)
124  {
125  switch (trigger)

128  EXTI->RTSR |= mask;

134  break;
135  case InputTrigger::BothEdges:

139  }
